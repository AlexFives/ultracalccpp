# Грамматика

```
ZERO ::= '0'
NOT_ZERO ::= [1-9]
DIGIT ::= ZERO | NOT_ZERO
DIGITS ::= DIGIT+

SIGN ::= '+' | '-'
POINT ::= '.'
EXP ::= 'e' | 'E'
I ::= 'i' | 'I'

N_NUMBER ::= DIGITS
Z_NUMBER ::= SIGN? N_NUMBER
FLOATING_POINT_NUMBER ::= Z_NUMBER (POINT DIGITS)?
EXP_NUMBER ::= FLOATING_POINT_NUMBER (EXP Z_NUMBER)?
IMAGINE_NUMBER ::= SIGN? EXP_NUMBER? I
C_NUMBER ::=
            IMAGINE_NUMBER (SIGN EXP_NUMBER)? |
            EXP_NUMBER (SIGN IMAGINE_NUMBER)?

```
